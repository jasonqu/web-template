var text = "I am exposed!";

function first(){
    var insideFirst = "I am inside first";
    function second(){
        var insideFirst = "I am inside second";
        console.log(text);        // I am exposed!
        console.log(insideFirst); // I am inside second
    }

    function third(text){
        insideThird = "I am inside third";
        console.log(text);        // into third
        console.log(insideFirst); // I am inside first
        console.log(insideThird); // I am inside third
    }
    third("into third");
}

function anotherFunc(){
    // variables
    console.log(text);        // I am exposed!
    console.log(insideThird); // I am inside third
    console.log(insideFirst); // Uncaught ReferenceError: insideFirst is not defined
    // functions
    first();
    second();
}

first();
anotherFunc();
